#Starts up a server
docker run -d --restart always --privileged --dns 8.8.8.8 --name node1 -h node1 --publish-all=true -d  --net=es_bridge --ip 172.18.0.2 --link db_node:postgres  -i -t elastic/server

#Starts up postgres
docker run --restart always --privileged --dns 8.8.8.8 --name db_node -h db_node --ip 172.18.0.20 --net=es_bridge -e POSTGRES_PASSWORD=password -d elastic/db_server


#From within a container, to connect to postgres using psql:
psql -h db_server -U postgres
